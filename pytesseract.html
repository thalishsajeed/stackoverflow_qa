def display_image(filename, length_box=60, width_box=30):
    if type(filename) == np.ndarray:
        image = filename
    else:
        image = cv2.imread(filename)
    plt.figure(figsize=(length_box, width_box))
    plt.imshow(image, cmap="gray")
